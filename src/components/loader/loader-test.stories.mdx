export const LoaderStory = ({ isInsideButton, size, isActive, ...args }) => {
  if (isInsideButton) {
    return (
      <Button buttonType="primary" disabled={!isActive}>
        <Loader {...args} />
      </Button>
    );
  }
  return <Loader size={size} />;
};
