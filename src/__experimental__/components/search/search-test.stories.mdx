import { useState } from "react";
import { Meta, Story, Preview } from "@storybook/addon-docs/blocks";
import { action } from "@storybook/addon-actions";

import Search from ".";

<Meta
  title="Design System/Search/Test"
  parameters={{
    info: { disable: true },
    chromatic: {
      disable: true,
    },
    knobs: {
      disable: true,
    },
  }}
  argTypes={{
    variant: {
      options: ["default", "dark"],
      control: {
        type: "select",
      },
    },
  }}
/>

export const SearchStory = (args) => {
  const [value, setValue] = useState("");
  const handleChange = (ev) => {
    setValue(ev.target.value);
    action("change")(ev);
  };
  const handleBlur = (ev) => {
    action("blur")(ev);
  };
  const handleClick = (ev) => {
    action("click")(ev);
  };
  return (
    <Search
      onChange={handleChange}
      onBlur={handleBlur}
      onClick={handleClick}
      value={value}
      name="search_name"
      id="search_id"
      {...args}
    />
  );
};
