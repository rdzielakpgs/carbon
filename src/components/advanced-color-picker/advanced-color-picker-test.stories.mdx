import { useState } from "react";
import { Meta, Story, Preview } from "@storybook/addon-docs/blocks";
import { action } from "@storybook/addon-actions";

import AdvancedColorPicker from ".";

<Meta
  title="Design System/Advanced Color Picker/Test"
  parameters={{
    info: { disable: true },
    chromatic: {
      disable: true,
    },
    knobs: {
      disable: true,
    },
  }}
/>

export const ActionPopoverStory = (args) => {
  const [state, setState] = useState({
    open: false,
    selectedColor: null,
  });
  const onOpen = (e) => {
    setState({
      open: true,
      selectedColor: state.selectedColor,
    });
    action("Open")(e);
  };
  const onClose = (e) => {
    setState({
      open: false,
      selectedColor: state.selectedColor,
    });
    action("Close")(e);
  };
  const onChange = (e) => {
    const { value } = e.target;
    setState({
      selectedColor: value,
    });
    action(`Selected - ${value}`)(e);
  };
  const onBlur = (e) => {
    action("Blur")(e);
  };
  return (
    <AdvancedColorPicker
      onOpen={onOpen}
      onClose={onClose}
      onChange={onChange}
      onBlur={onBlur}
      selectedColor={state.selectedColor}
      defaultColor="#EBAEDE"
      open={state.open}
      {...args}
    />
  );
};
